<script>
            document.addEventListener('DOMContentLoaded', function() {

                var is_chrome = navigator.userAgent.toLowerCase().indexOf('chrome') > -1;

                if(is_chrome) {
                    if(Notification.permission !== 'granted') {
                        Notification.requestPermission().then(function(permission) {
                            if(permission === 'granted' && 'serviceWorker' in navigator) {
                                navigator.serviceWorker.register('/sw.js?v=13042019-1358').then(initialiseState);
                            } else {
                                console.log('service worker not present');
                            }
                        });
                    }
                }
            });

            function initialiseState() {

                //check if notification is supported or not
                if(!('showNotification' in ServiceWorkerRegistration.prototype)) {
                    console.log('Notificaiton are not supported');
                    return;
                }
                //check if user has blocked push notification
                if(Notification.permission === 'denied'){
                    console.log('User has blocked the notification');
                }
                //check if push messaging is supported or not
                if(!('PushManager' in window)) {
                    console.log('Push messaging is not supported');
                    return;
                }

                navigator.serviceWorker.ready
                    .then(function(serviceWorkerRegistration) {
                        return serviceWorkerRegistration.pushManager.subscribe({
                            userVisibleOnly: true
                        });
                    })
                    .then(function(subscription) {
                        var token = subscription.endpoint.substring(40, subscription.endpoint.length);
                        fetch('/web_tools/notificaciones', {
                            method: 'POST',
                            headers: {
                                "Content-Type": "application/json"
                            },
                            body: JSON.stringify({'user_token': token, '_token': window.laravel_token})
                        }).then(function(response) {
                            return response.json();
                        }).then(function(myJson) {
                            console.log(myJson);
                        });
                    });
            }
        </script>
